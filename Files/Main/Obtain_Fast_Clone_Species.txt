; Box 10 Slot 19

SBC r11, r15, #0x2940      ; r11 = &Box10Slot19 + 7 (on emulator)
LDRH r12, [r11, #0x15]     ; 0x13 on console
ADC r12, r12, #0x2840
SBC r12, r12, #0xdf
STRH r12, [r11, #0x15]     ; Write checksum, 0x13 on console
MOVS r12, #0xbc00
ADC r12, r12, #0x348
STRH r12, [r11, #0x25]     ; Write species, 0x23 on console
