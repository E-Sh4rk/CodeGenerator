@@ exit = "CertificateENG"

movs r12, #0x20244ec ? ; r12=Pokemon_party

ldrh r11, [r12, #0xac]  ; Load lowIV
ldrh r10, [r12, #0xae]  ; Load highIV

strh r11, [r12, #0xBE] ; Store lowTID in Attack stat
strh r10, [r12, #0xC0] ; Store highTID in Defense stat
