@@ title = "Read TID"
@@ author = "E-Sh4rk"
@@ exit = "CertificateENG"
@@

movs r12, #0x020244D0 ? ; R12=Party_offset+0x64-0x80

ldrh r11, [r12, #0x84]  ; Load lowTID
ldrh r10, [r12, #0x86]  ; Load highTID

strh r11, [r12, #0xDA] ; Store lowTID in Attack stat
strh r10, [r12, #0xDC] ; Store highTID in Defense stat
